```typescript
 */
export interface PackageTemplate {
  id: string;
  name: string;                        // Template name (can include {language})
  description: string;                 // Template description
  category: PackageCategory;
  
  // Requirements for this template
  requirements: {
    // Resource requirements (checked against Door43 catalog)
    resources: ResourceRequirement[];
    
    // Owner requirement (which organization)
    owners: string[];                  // e.g., ['unfoldingWord', 'Door43-Catalog']
    
    // Subject requirements (from Door43 catalog)
    subjects?: string[];               // e.g., ['Bible', 'Aligned Bible']
  };
  
  // Package configuration when generated
  config: {
    offlineEnabled: boolean;
    autoUpdate: boolean;
  };
  
  // Panel layout
  panelLayout: PanelLayoutTemplate[];
}
```

```typescript
// lib/services/packages/DefaultPackageGenerator.ts

export class DefaultPackageGenerator {
  constructor(
    private discoveryService: ResourceDiscoveryService,
    private templates: PackageTemplate[] = DEFAULT_PACKAGE_TEMPLATES
  ) {}
  
  /**
   * Generate all available default packages
   */
  async generateDefaultPackages(): Promise<DefaultPackage[]> {
    // Step 1: Get all available languages from Door43
    const languages = await this.discoveryService.getAvailableLanguages();
    
    const packages: DefaultPackage[] = [];
    
    // Step 2: For each language, check which templates can be fulfilled
    for (const language of languages) {
      for (const template of this.templates) {
        const pkg = await this.tryGeneratePackage(template, language);
        if (pkg) {
          packages.push(pkg);
        }
      }
    }
    
    // Step 3: Sort by language name, then by category
    packages.sort((a, b) => {
      if (a.languageName !== b.languageName) {
        return a.languageName.localeCompare(b.languageName);
      }
      return a.category.localeCompare(b.category);
    });
    
    return packages;
  }
  
  /**
   * Try to generate a package from a template for a specific language
   */
  private async tryGeneratePackage(
    template: PackageTemplate,
    language: LanguageInfo
  ): Promise<DefaultPackage | null> {
    const resources: GeneratedResource[] = [];
    
    // Check each resource requirement
    for (const req of template.requirements.resources) {
      const lang = req.languageOverride || language.code;
      
      // Try each resource ID in priority order
      let found = false;
      for (const resourceId of req.resourceIds) {
        const available = await this.checkResourceAvailable(
          resourceId,
          lang,
          template.requirements.owners
        );
        
        if (available) {
          resources.push({
            resourceId,
            language: lang,
            type: req.type,
            role: req.role,
            owner: available.owner,
            version: available.version
          });
          found = true;
          break;
        }
      }
      
      // If required resource not found, template doesn't apply
      if (!found && req.required) {
        return null;
      }
    }
    
    // All required resources found - generate package
    return {
      id: `default-${template.id}-${language.code}`,
      templateId: template.id,
      name: template.name.replace('{language}', language.name),
      description: template.description,
      category: template.category,
      language: language.code,
      languageName: language.name,
      languageDirection: language.direction,
      resources,
      config: template.config,
      panelLayout: template.panelLayout,
      isDefault: true,
      generatedAt: new Date()
    };
  }
  
  /**
   * Get packages for a specific language
   */
  async getPackagesForLanguage(languageCode: string): Promise<DefaultPackage[]> {
    const allPackages = await this.generateDefaultPackages();
    return allPackages.filter(pkg => pkg.language === languageCode);
  }
  
  /**
   * Get packages by category
   */
  async getPackagesByCategory(category: PackageCategory): Promise<DefaultPackage[]> {
    const allPackages = await this.generateDefaultPackages();
    return allPackages.filter(pkg => pkg.category === category);
  }
}
```
